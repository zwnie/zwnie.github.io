cmake阅读记录

发表于 2021-11-01 更新于 2023-06-28 分类于 Gromacs
本文字数： 4.5k 阅读时长 ≈ 4 分钟

CMakeLists.txt

cmake是makefile的上层工具，用以产生可移植的Makefile，并简化工作量。CMakeLists.txt是cmake的构建配置文件：

list(APPEND CMAKE_MODULE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/cmak    ${CMAKE_CURRENT_SOURCE_DIR}/cmake/Platform)

扩充了CMAKE_MODULE_PATH路径，随后我们调用的模块会在相应路径下搜寻*.cmake文件；

project(Gromacs)

项目的名称是Gromacs；

set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

设定了动态库、静态库以及可执行文件的build路径；

option(GMX_MPI    "Build a parallel (message-passing) version of GROMACS" OFF)

后面一大部分，利用option读取用户输入，构建相应变量。以GMX_MPI为例，-DGMX_MPI=on，如果未指定，默认值off；

 

再往后一大段都是外部package的处理，无非是find_package - include_diractories，因为我们使用默认值编译，所以没影响；

再后面一大段是一些对硬件、系统管线等的测试，影响不大；接下来是FFT库的部分：

include(gmxManageFFTLibraries)

决定FFT库的单精度还是双精度，添加了include头文件搜索路径等；

include(gmxManageLinearAlgebraLibraries)

判断BLAS, LAPACK是否由用户指定，否则compile internal，库的路径保存在LINEAR_ALGEBRA_LIBRARIES；

之后又定义了一堆诸如GMX_PUBLIC_LIBRARIES之类的变量；

if (NOT GMX_SKIP_DEFAULT_CFLAGS)
    set(CMAKE_EXE_LINKER_FLAGS "${FFT_LINKER_FLAGS} ${MPI_LINKER_FLAGS} ${CMAKE_EXE_LINKER_FLAGS} ${DISABLE_SYCL_CXX_FLAGS}")
    set(CMAKE_SHARED_LINKER_FLAGS "${FFT_LINKER_FLAGS} ${MPI_LINKER_FLAGS} ${CMAKE_SHARED_LINKER_FLAGS} ${DISABLE_SYCL_CXX_FLAGS}")

最后汇总前面的标签，产生了CMAKE_EXE_LINKER_FLAGS， CMAKE_SHARED_LINKER_FLAGS两个用来链接可执行文件和共享库的标签；

install(FILES COPYING DESTINATION ${GMX_INSTALL_GMXDATADIR} COMPONENT data)

漫长cmake文件中第一个install指令(ಥ_ಥ) 具体是将${CMAKE_SOURCE_DIR}下的COPYING文件安装到${CMAKE_INSTALL_PREFIX}/${GMX_INSTALL_GMXDATADIR}下，在make - make install后能在./my-install/share/gromacs找到；

if (BUILD_TESTING)
    include(tests/CheckTarget.cmake)
endif()

include CheckTarget.cmake文件，看了下，具体内容是调用ctest构建了make check等一系列指令，前提是有一个名为test的target，test具体是什么还没定义；

我们忽略用不上的docs,share,scripts,api等，关注src目录：

add_subdirectory(src)

src/gromacs/CMakelist.txt

它里面又分为gromacs,programs,api三个部分，首先关注gromacs

add_subdirectory(gromacs)

里面定义了一个变量

set(LIBGROMACS_SOURCES)

后续所有的源文件都会放进去；

里面还定义了gmx_add_libgromacs_sources函数

function (_gmx_add_files_to_property PROPERTY)
    foreach (_file ${ARGN})
        if (IS_ABSOLUTE "${_file}")
            set_property(GLOBAL APPEND PROPERTY ${PROPERTY} ${_file})
        else()
            set_property(GLOBAL APPEND PROPERTY ${PROPERTY}
                        ${CMAKE_CURRENT_LIST_DIR}/${_file})
        endif()
    endforeach()
endfunction ()

function (gmx_add_libgromacs_sources)
    _gmx_add_files_to_property(GMX_LIBGROMACS_SOURCES ${ARGN})
endfunction ()

可以看到，具体作用是设定了一个名为GMX_LIBGROMACS_SOURCES的全局的特性，会将ARGN里面所有文件的绝对路径都添加进去，这个函数被反复调用，用以储存所有源文件。

get_property(PROPERTY_SOURCES GLOBAL PROPERTY GMX_LIBGROMACS_SOURCES)
list(APPEND LIBGROMACS_SOURCES ${GMXLIB_SOURCES} ${MDLIB_SOURCES} ${PROPERTY_SOURCES})

最终，所有的源文件都被放到LIBGROMACS_SOURCES里边。

target_link_libraries(libgromacs
                    PRIVATE
                    ${EXTRAE_LIBRARIES}
                    ${GMX_EXTRA_LIBRARIES}
                    ${GMX_COMMON_LIBRARIES}
                    ${FFT_LIBRARIES} ${LINEAR_ALGEBRA_LIBRARIES}
                    ${THREAD_LIB} ${GMX_SHARED_LINKER_FLAGS}
                    ${SYCL_CXX_FLAGS}
                    ${OpenCL_LIBRARIES}
                    $<$<PLATFORM_ID:SunOS>:socket>
                    PUBLIC
                    ${GMX_PUBLIC_LIBRARIES}
                    )

将需要用到的的共享库链接到libgromacs库；

set_target_properties(libgromacs PROPERTIES
                    OUTPUT_NAME "gromacs${GMX_LIBS_SUFFIX}"
                    SOVERSION ${LIBRARY_SOVERSION_MAJOR}
                    VERSION ${LIBRARY_VERSION}
                    )

设定了输出名称:gmx_mpi，及版本信息；

install(TARGETS libgromacs
        EXPORT libgromacs
        LIBRARY
            DESTINATION ${CMAKE_INSTALL_LIBDIR}
            COMPONENT libraries
        RUNTIME
            DESTINATION ${CMAKE_INSTALL_BINDIR}
            COMPONENT libraries
        ARCHIVE
            DESTINATION ${CMAKE_INSTALL_LIBDIR}
            COMPONENT libraries
        INCLUDES DESTINATION include)
target_compile_definitions(libgromacs PUBLIC $<INSTALL_INTERFACE:GMX_DOUBLE=${GMX_DOUBLE_VALUE}>)

指定了动态库、静态库以及可执行文件的安装路径，add_library(libgromacs)是动态库，所以只会有动态库安装到lib目录下；

src/programs/CMakelist.txt

add_subdirectory(programs)
...
add_executable(gmx
    $<TARGET_OBJECTS:gmx_objlib>
    $<TARGET_OBJECTS:mdrun_objlib>
    $<TARGET_OBJECTS:view_objlib>)
...
target_link_libraries(gmx libgromacs
    ${GMX_COMMON_LIBRARIES}
    ${GMX_EXE_LINKER_FLAGS})
...
set(BINARY_NAME "gmx${GMX_BINARY_SUFFIX}")
set_target_properties(gmx PROPERTIES
    OUTPUT_NAME "${BINARY_NAME}")

定义了这样3个库：mdrun_objlib, view_objlib, gmx_objlib，共同用他们生成了名为gmx${GMX_BINARY_SUFFIX}的可执行文件，链接上上一步生成的libgromacs动态库，最后被安装到bin目录下；
